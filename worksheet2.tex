% Lining up several equations.
\begin{align*}
  (a + b) + (c + d) &= ((a + b) + c) + d \\
  &= (a + (b + c)) + d \\ 
  &= a + ((b + c) + d) \\ 
  &= a + (b + (c + d)) 
\end{align*}

% Numbering an equation
\begin{align}
	||a+b|| \leq ||a|| + ||b||
\end{align}

\]
